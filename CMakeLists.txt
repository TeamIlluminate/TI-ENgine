cmake_minimum_required(VERSION 3.0.0)
project(TI-Engine VERSION 0.1.0)
enable_testing()
aux_source_directory(src/Components CMP)
aux_source_directory(src SRC)
aux_source_directory(src/Game GAM)
include_directories(src)
add_executable(main src/main.cpp ${SRC} ${CMP} ${GAM})
target_link_libraries (main -lsfml-graphics -lsfml-window -lsfml-system -lBox2D)
set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)
